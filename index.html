				/*
				 * footnotes jQuery plugin
				 * Scan all specified elements (links by default) in a specified DOM element (usually body) in order to put footnotes for print purposes.
				 *
				 * Settings :
				 {
					insertInsideSel: this.selector, // jQuery selector : where the footnotes list is insert in.
					linksSel: 'a', // jQuery selector : selector to match DOM elements within the current scope
					footNoteClass: 'footnote', // string class : class attribute for all matched DOM elements
					printOnlyClass: 'printOnly', // string class : class attribute for elements to show only for a print media.
					footNotesID: 'footnotes', // string ID : identifier for the footnotes division
					scanAnchors: false, // Boolean true/false : anchors only (start with #) are checked for double matchs
					scanTitle: false, // Boolean true/false : titles are checked for double matchs
					showTitle: false, // Boolean true/false : show title in footnotes list
					scanTarget: false, // Boolean true/false : targets are checked for double matchs
					showTarget: false // Boolean true/false : show target in footnotes list
				 }
				 *
				 */
